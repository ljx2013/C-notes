
# C++ 文件操作

## 一、概述

文件操作是程序与外部存储设备交互的重要方式。C++ 提供了两种文件操作方式：

1. **C 风格**：使用 `<cstdio>` 中的函数
2. **C++ 风格**：使用 `<fstream>` 中的流类

## 二、C++ 风格文件操作

### 2.1 头文件

```cpp
#include <fstream>
```

### 2.2 文件流类型

| 类型 | 用途 |
|------|------|
| `ifstream` | 读取文件 |
| `ofstream` | 写入文件 |
| `fstream` | 读写文件 |

### 2.3 文件打开模式

| 模式 | 说明 |
|------|------|
| `ios::in` | 读取模式 |
| `ios::out` | 写入模式 |
| `ios::app` | 追加模式 |
| `ios::trunc` | 截断模式 |
| `ios::binary` | 二进制模式 |
| `ios::ate` | 定位到文件末尾 |

## 三、文本文件操作

### 3.1 写入文本文件

```cpp
#include <fstream>
#include <iostream>
using namespace std;

int main() {
    ofstream outfile("example.txt");
    
    if (outfile.is_open()) {
        outfile << "Hello, World!" << endl;
        outfile << "This is a test file." << endl;
        outfile.close();
    } else {
        cout << "Unable to open file" << endl;
    }
    
    return 0;
}
```

### 3.2 读取文本文件

```cpp
#include <fstream>
#include <iostream>
#include <string>
using namespace std;

int main() {
    ifstream infile("example.txt");
    string line;
    
    if (infile.is_open()) {
        while (getline(infile, line)) {
            cout << line << endl;
        }
        infile.close();
    } else {
        cout << "Unable to open file" << endl;
    }
    
    return 0;
}
```

### 3.3 逐词读取

```cpp
#include <fstream>
#include <iostream>
#include <string>
using namespace std;

int main() {
    ifstream infile("example.txt");
    string word;
    
    while (infile >> word) {
        cout << word << endl;
    }
    
    return 0;
}
```

## 四、二进制文件操作

### 4.1 写入二进制文件

```cpp
#include <fstream>
using namespace std;

struct Person {
    char name[50];
    int age;
    double height;
};

int main() {
    Person p = {"Alice", 25, 165.5};
    
    ofstream outfile("person.dat", ios::binary);
    if (outfile) {
        outfile.write(reinterpret_cast<char*>(&p), sizeof(p));
        outfile.close();
    }
    
    return 0;
}
```

### 4.2 读取二进制文件

```cpp
#include <fstream>
#include <iostream>
using namespace std;

struct Person {
    char name[50];
    int age;
    double height;
};

int main() {
    Person p;
    
    ifstream infile("person.dat", ios::binary);
    if (infile) {
        infile.read(reinterpret_cast<char*>(&p), sizeof(p));
        cout << "Name: " << p.name << endl;
        cout << "Age: " << p.age << endl;
        cout << "Height: " << p.height << endl;
        infile.close();
    }
    
    return 0;
}
```

## 五、文件定位

### 5.1 seekg 和 seekp

```cpp
#include <fstream>
#include <iostream>
using namespace std;

int main() {
    fstream file("example.txt", ios::in | ios::out);
    
    if (file) {
        // 定位到第10个字节
        file.seekg(10);
        
        // 获取当前位置
        cout << "Current position: " << file.tellg() << endl;
        
        // 定位到文件末尾
        file.seekg(0, ios::end);
        
        // 获取文件大小
        int size = file.tellg();
        cout << "File size: " << size << endl;
        
        file.close();
    }
    
    return 0;
}
```

### 5.2 定位模式

| 模式 | 说明 |
|------|------|
| `ios::beg` | 从文件开头 |
| `ios::cur` | 从当前位置 |
| `ios::end` | 从文件末尾 |

## 六、文件状态检查

### 6.1 常用状态函数

```cpp
#include <fstream>
using namespace std;

int main() {
    ifstream file("example.txt");
    
    if (file.is_open()) {
        // 检查状态
        bool good = file.good();   // 状态正常
        bool eof = file.eof();     // 到达文件末尾
        bool fail = file.fail();   // 操作失败
        bool bad = file.bad();     // 严重错误
    }
    
    return 0;
}
```

## 七、C 风格文件操作

### 7.1 打开和关闭文件

```cpp
#include <cstdio>

int main() {
    FILE* file = fopen("example.txt", "w");
    
    if (file != nullptr) {
        fprintf(file, "Hello, World!\n");
        fclose(file);
    }
    
    return 0;
}
```

### 7.2 文件访问模式

| 模式 | 说明 |
|------|------|
| `"r"` | 只读 |
| `"w"` | 写入（创建或截断） |
| `"a"` | 追加 |
| `"rb"` | 二进制只读 |
| `"wb"` | 二进制写入 |
| `"ab"` | 二进制追加 |

### 7.3 读写操作

```cpp
#include <cstdio>

int main() {
    // 写入
    FILE* out = fopen("data.txt", "w");
    if (out) {
        fprintf(out, "%d %s %.2f\n", 42, "hello", 3.14);
        fclose(out);
    }
    
    // 读取
    FILE* in = fopen("data.txt", "r");
    if (in) {
        int num;
        char str[100];
        double val;
        fscanf(in, "%d %s %lf", &num, str, &val);
        fclose(in);
    }
    
    return 0;
}
```

## 八、文件操作实战

### 示例1：文件复制

```cpp
#include <fstream>
#include <iostream>
using namespace std;

bool copyFile(const string& source, const string& dest) {
    ifstream src(source, ios::binary);
    ofstream dst(dest, ios::binary);
    
    if (!src || !dst) {
        return false;
    }
    
    dst << src.rdbuf();
    return true;
}

int main() {
    if (copyFile("source.txt", "dest.txt")) {
        cout << "File copied successfully" << endl;
    } else {
        cout << "Failed to copy file" << endl;
    }
    return 0;
}
```

### 示例2：统计文件行数

```cpp
#include <fstream>
#include <iostream>
#include <string>
using namespace std;

int countLines(const string& filename) {
    ifstream file(filename);
    if (!file) return -1;
    
    int count = 0;
    string line;
    while (getline(file, line)) {
        count++;
    }
    return count;
}

int main() {
    int lines = countLines("example.txt");
    if (lines >= 0) {
        cout << "Number of lines: " << lines << endl;
    } else {
        cout << "Unable to open file" << endl;
    }
    return 0;
}
```

### 示例3：读取 CSV 文件

```cpp
#include <fstream>
#include <iostream>
#include <sstream>
#include <vector>
#include <string>
using namespace std;

vector<string> split(const string& s, char delimiter) {
    vector<string> tokens;
    string token;
    istringstream tokenStream(s);
    while (getline(tokenStream, token, delimiter)) {
        tokens.push_back(token);
    }
    return tokens;
}

int main() {
    ifstream file("data.csv");
    string line;
    
    while (getline(file, line)) {
        vector<string> fields = split(line, ',');
        for (const string& field : fields) {
            cout << field << "\t";
        }
        cout << endl;
    }
    
    return 0;
}
```

## 九、总结

| 操作 | C++ 风格 | C 风格 |
|------|----------|--------|
| 打开文件 | `ofstream`, `ifstream` | `fopen` |
| 写入 | `<<` 操作符 | `fprintf`, `fwrite` |
| 读取 | `>>` 操作符, `getline` | `fscanf`, `fread` |
| 关闭 | `close()` | `fclose` |
| 定位 | `seekg`, `seekp` | `fseek` |
| 状态 | `good()`, `eof()`, `fail()` | `feof`, `ferror` |

文件操作是程序的重要组成部分，掌握 C++ 风格的文件流操作是编写实用程序的必备技能。
